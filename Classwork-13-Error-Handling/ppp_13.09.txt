
    TRY
        DISPLAY "Verb to conjugate: "
        READ input_verb
        verb = CONVERT_TO_LOWERCASE(TRIM_SPACES(input_verb))
        
        IF verb EQUALS "" THEN
            THROW NEW ValueError("You must enter a verb.")
        END IF
        
        IF verb DOES NOT CONTAIN ONLY LETTERS THEN
            THROW NEW ValueError("The verb must contain only letters.")
        END IF
        
        last_two_letters = SUBSTRING(verb, FROM -2 TO END)
        IF last_two_letters IS NOT IN ["ar", "er", "ir"] THEN
            THROW NEW InvalidEndingError("The verb must end in 'ar', 'er', or 'ir'.")
        END IF

        pronouns = ['Yo', 'Tu', 'El', 'Nosotros', 'Vosotros', 'Ellos']

        CREATE dictionary endings
        endings['ar'] = ['o', 'as', 'a', 'amos', 'ais', 'an']
        endings['er'] = ['o', 'es', 'e', 'emos', 'eis', 'en']
        endings['ir'] = ['o', 'es', 'e', 'imos', 'is', 'en']

        stem = SUBSTRING(verb, FROM START TO -2)
        ending = SUBSTRING(verb, FROM -2 TO END)
        ending_list = endings[ending]

        FOR EACH index, pronoun IN ENUMERATE(pronouns) DO
            end = ending_list[index]
            DISPLAY pronoun + " " + stem + end
        END FOR

    CATCH InvalidEndingError AS e
        DISPLAY e.message
    CATCH ValueError AS e
        DISPLAY e.message
    END TRY
